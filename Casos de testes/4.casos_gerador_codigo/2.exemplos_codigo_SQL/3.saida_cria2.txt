CREATE TABLE Funcionario {
    nome VARCHAR(255),
    idade INT,
    cpf INT,
    profissao TEXT
}

CREATE TABLE Aluno {
    nome VARCHAR(255),
    idade INT,
    ra INT,
    curso VARCHAR(255)
}

   WHERE id
   AND nome = "joao"
